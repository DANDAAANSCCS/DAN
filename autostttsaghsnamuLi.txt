-- FunciÃ³n para aplicar stats (FORZADO - aplica simultÃ¡neamente)
local function aplicarStats(stat, cantidad, velocidad)
	local config = Velocidades[velocidad]
	local aplicados = 0
	
	print("â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•")
	print("ğŸ“Š Aplicando " .. tostring(cantidad) .. " puntos a " .. stat)
	print("âš¡ Velocidad: " .. config.nombre)
	print("â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•â•")
	
	getgenv().AutoStatsRunning = true
	
	while aplicados < cantidad and getgenv().AutoStatsRunning do
		local puntosEnBloque = math.min(config.bloques, cantidad - aplicados)
		
		-- FORZADO: Aplicar todos los del bloque SIMULTÃNEAMENTE con spawn
		for i = 1, puntosEnBloque do
			if not getgenv().AutoStatsRunning then break end
			
			spawn(function()
				pcall(function()
					ReplicatedStorage.Events.UpgradeAbility:InvokeServer(stat)
				end)
			end)
			
			aplicados = aplicados + 1
		end
		
		-- Mostrar progreso cada 50 puntos
		if aplicados % 50 == 0 or aplicados == cantidad then
			print("   ğŸ“ˆ Progreso: " .. tostring(aplicados) .. "/" .. tostring(cantidad) .. " (" .. math.floor(aplicados/cantidad*100) .. "%)")
		end
		
		task.wait(config.espera)
	end
	
	if aplicados >= cantidad then
		print("âœ… Â¡Completado! " .. tostring(aplicados) .. " puntos aplicados a " .. stat)
	else
		print("â¹ï¸ Detenido. " .. tostring(aplicados) .. " puntos aplicados a " .. stat)
	end
	
	getgenv().AutoStatsRunning = false
end

-- FunciÃ³n Anti-AFK
local function iniciarAntiAFK()
	-- Desconectar eventos de Idled
	for _, connection in pairs(getconnections(player.Idled)) do
		connection:Disable()
	end
	
	-- Reconectar con nuestro handler
	player.Idled:Connect(function()
		if getgenv().AntiAFKEnabled then
			VirtualUser:Button2Down(Vector2.new(0, 0), workspace.CurrentCamera.CFrame)
			task.wait(1)
			VirtualUser:Button2Up(Vector2.new(0, 0), workspace.CurrentCamera.CFrame)
			print("ğŸ”„ Anti-AFK activado - Evitando kick")
		end
	end)
end
